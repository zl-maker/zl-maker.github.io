通用寄存器模块源码

`timescale 1ns / 1ps

module REG(clk,rst,Write_Reg,Write_PC,R_Addr_A,R_Addr_B,R_Addr_C,W_Addr,W_Data,PC_New,M,R_Data_A,R_Data_B,R_Data_C);
	input clk;//写入时钟信号
    input rst;//清零信号
    input Write_Reg;//写控制信号
    input Write_PC;
    input [4:0]R_Addr_A;
    input [4:0]R_Addr_B;
    input [4:0]R_Addr_C;
    input [4:0]W_Addr;
    input [31:0]W_Data;
    input [31:0]PC_New;
    input [4:0]M;
	output reg[31:0]R_Data_A;
    output reg[31:0]R_Data_B;
    output reg[31:0]R_Data_C;
	 
	integer i;
	integer Error1;
	integer Error2;
	//0~14 表示通用寄存器
	//随后是fiq模式的8~14
	//最后是每个模式的13~14
	reg [34:0] R[31:0];
	  
    initial
        for(i=0;i<33;i=i+1) R[i]<=0;
    
    // 对寄存器进行写操作
    always@(posedge clk or posedge rst)
   	 begin
        if(rst)
                for(i=0;i<34;i=i+1) R[i]<=0;
        else
            begin
             if(M[4]==15&&W_Addr==15) Error1=1;
             else
             begin
                Error1=0;
                case(M[3:0])
                 4'b0000,4'b1111:begin
                  if(Write_Reg)
                   begin
                    R[W_Addr]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                 4'b0001:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<8) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+8]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                 4'b0010:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+10]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end                 
                 4'b0011:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+12]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                 4'b0110:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+14]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                 4'b0111:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+16]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                 4'b1010:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+18]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end               
                 4'b1011:begin
                  if(Write_Reg)
                   begin
                    if(W_Addr<12) R[W_Addr]<=W_Data;
                    else if(W_Addr<15) R[W_Addr+20]<=W_Data;
                   end    
                  if(Write_PC)
                   begin
                    R[15]<=PC_New;
                   end
                 end
                endcase
             end
  	        end
    end
    
    //对寄存器进行读操作
    always@(*)
    begin
        Error2=0;
        //读取A
        if(R_Addr_A<8)
         begin
          R_Data_A<=R[R_Addr_A];
         end
        else if(R_Addr_A<13)
         begin
          if(M[3:0]==4'b0001) R_Data_A<=R[R_Addr_A+8];
          else if(M[3:0]==4'b0000||M[3:0]==4'b0100||M[3:0]==4'b0101||M[3:0]==4'b1000||M[3:0]==4'b1001||M[3:0]==4'b1100||M[3:0]==4'b1101||M[3:0]==4'b1110||M[3:0]==4'b1111)
           R_Data_A<=R[R_Addr_A];
          else Error2=1;
         end
        else if(R_Addr_A==13)
         begin
          case(M[3:0])
           4'b0000:R_Data_A<=R[13];
           4'b0001:R_Data_A<=R[21];
           4'b0010:R_Data_A<=R[23];
           4'b0011:R_Data_A<=R[25];
           4'b0110:R_Data_A<=R[27];
           4'b0111:R_Data_A<=R[29];
           4'b1010:R_Data_A<=R[31];
           4'b1011:R_Data_A<=R[33];
           4'b1111:R_Data_A<=R[13];
           default:Error2=1;
          endcase
         end
        else if(R_Addr_A==14)
         begin
          case(M[3:0])
           4'b0000:R_Data_A<=R[14];
           4'b0001:R_Data_A<=R[22];
           4'b0010:R_Data_A<=R[24];
           4'b0011:R_Data_A<=R[26];
           4'b0110:R_Data_A<=R[28];
           4'b0111:R_Data_A<=R[30];
           4'b1010:R_Data_A<=R[32];
           4'b1011:R_Data_A<=R[34];
           4'b1111:R_Data_A<=R[14];
           default:Error2=1;
          endcase         
         end
        else
          R_Data_A<=R[15];
        
        //读取B  
        if(R_Addr_B<8)
         begin
          R_Data_B<=R[R_Addr_B];
         end
        else if(R_Addr_B<13)
         begin
          if(M[3:0]==4'b0001) R_Data_B<=R[R_Addr_B+8];
          else if(M[3:0]==4'b0000||M[3:0]==4'b0100||M[3:0]==4'b0101||M[3:0]==4'b1000||M[3:0]==4'b1001||M[3:0]==4'b1100||M[3:0]==4'b1101||M[3:0]==4'b1110||M[3:0]==4'b1111)
           R_Data_B<=R[R_Addr_B];
          else Error2=1;
         end
        else if(R_Addr_B==13)
         begin
          case(M[3:0])
           4'b0000:R_Data_B<=R[13];
           4'b0001:R_Data_B<=R[21];
           4'b0010:R_Data_B<=R[23];
           4'b0011:R_Data_B<=R[25];
           4'b0110:R_Data_B<=R[27];
           4'b0111:R_Data_B<=R[29];
           4'b1010:R_Data_B<=R[31];
           4'b1011:R_Data_B<=R[33];
           4'b1111:R_Data_B<=R[13];
           default:Error2=1;
          endcase
         end
        else if(R_Addr_B==14)
         begin
          case(M[3:0])
           4'b0000:R_Data_B<=R[14];
           4'b0001:R_Data_B<=R[22];
           4'b0010:R_Data_B<=R[24];
           4'b0011:R_Data_B<=R[26];
           4'b0110:R_Data_B<=R[28];
           4'b0111:R_Data_B<=R[30];
           4'b1010:R_Data_B<=R[32];
           4'b1011:R_Data_B<=R[34];
           4'b1111:R_Data_B<=R[14];
           default:Error2=1;
          endcase         
         end
        else
          R_Data_B<=R[15];
        
        //读取C  
        if(R_Addr_C<8)
         begin
          R_Data_C<=R[R_Addr_C];
         end
        else if(R_Addr_C<13)
         begin
          if(M[3:0]==4'b0001) R_Data_C<=R[R_Addr_C+8];
          else if(M[3:0]==4'b0000||M[3:0]==4'b0100||M[3:0]==4'b0101||M[3:0]==4'b1000||M[3:0]==4'b1001||M[3:0]==4'b1100||M[3:0]==4'b1101||M[3:0]==4'b1110||M[3:0]==4'b1111)
           R_Data_C<=R[R_Addr_C];
          else Error2=1;
         end
        else if(R_Addr_C==13)
         begin
          case(M[3:0])
           4'b0000:R_Data_C<=R[13];
           4'b0001:R_Data_C<=R[21];
           4'b0010:R_Data_C<=R[23];
           4'b0011:R_Data_C<=R[25];
           4'b0110:R_Data_C<=R[27];
           4'b0111:R_Data_C<=R[29];
           4'b1010:R_Data_C<=R[31];
           4'b1011:R_Data_C<=R[33];
           4'b1111:R_Data_C<=R[13];
           default:Error2=1;
          endcase
         end
        else if(R_Addr_C==14)
         begin
          case(M[3:0])
           4'b0000:R_Data_C<=R[14];
           4'b0001:R_Data_C<=R[22];
           4'b0010:R_Data_C<=R[24];
           4'b0011:R_Data_C<=R[26];
           4'b0110:R_Data_C<=R[28];
           4'b0111:R_Data_C<=R[30];
           4'b1010:R_Data_C=R[32];
           4'b1011:R_Data_C<=R[34];
           4'b1111:R_Data_C<=R[14];
           default:Error2=1;
          endcase         
         end
        else
          R_Data_C<=R[15];          
    
    end

endmodule

module simu;
	reg clk;
    reg rst;
    reg Write_Reg;
    reg Write_PC;
    reg [4:0]R_Addr_A;
    reg [4:0]R_Addr_B;
    reg [4:0]R_Addr_C;
    reg [4:0]W_Addr;
    reg [31:0]W_Data;
    reg [31:0]PC_New;
    reg [4:0]M;
    
	wire [31:0]R_Data_A;
    wire [31:0]R_Data_B;
    wire [31:0]R_Data_C;

    REG uut(clk,rst,Write_Reg,Write_PC,R_Addr_A,R_Addr_B,R_Addr_C,W_Addr,W_Data,PC_New,M,R_Data_A,R_Data_B,R_Data_C);

    initial begin
    
       //0
       M=5'b10000;
       W_Addr=1;
       W_Data=32'b00000000000000000000000000000001;
       Write_Reg=1;
       #50
       clk=1;
       //100
       #50
       clk=0;
       W_Addr=2;
       W_Data=32'b00000000000000000000000000000010;
       Write_Reg=1;       
       #50
       clk=1;

       #50
       clk=0;

       R_Addr_A=1;
